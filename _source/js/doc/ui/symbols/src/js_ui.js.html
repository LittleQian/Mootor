 10 (function ($) {
 11 
 12  "use strict";
 13 
 14      /**
 15      * Checkbox instance object
 16      *
 17      * @class
 18      * @name Checkbox
 19      * @return {Checkbox} Checkbox object
 20      * @param {object} options  Configuration options
 21      * @property {element} el Container element
 22      */
 23     var Checkbox = function(options) {
 24         var check,
 25             input;
 26 
 27         this.el = options.el;
 28         this.el.innerHTML += "<b></b>";
 29 
 30         // FIXME CHECK: update core to support 'each'
 31         input = this.el.getElementsByTagName('input')[0];
 32         // FIXME CHECK: temporary initial value
 33         this.value = 1;
 34         $(input).hide();
 35         this.input = input;
 36 
 37         check = $(this.el);
 38         check.setClass("moo_checkbox");
 39 
 40         // FIXME CHECK: optimize me (initial value)
 41         if (input.checked === false) {
 42             this.toggle();
 43         }
 44         this.handleGesture = function (gesture) {
 45             this.toggle();
 46         };
 47         check.onTapEnd(this);
 48 
 49         return this;
 50     };
 51 
 52     Checkbox.prototype = {
 53      /**
 54      * Toggle control
 55      *
 56      * @this {Checkbox}
 57      * @example var myCheck =  $("#checkbox1").checkbox();
 58      * myCheck.toggle();
 59      */
 60      toggle: function () {
 61             var el = $(this.el);
 62             if (this.value === 0) {
 63                 this.value = 1;
 64                 this.input.checked = true;
 65                 el.removeClass("off");
 66                 el.setClass("on");
 67             } else {
 68                 this.input.checked = false;
 69                 el.removeClass("on");
 70                 el.setClass("off");
 71                 this.value = 0;
 72             }
 73         }
 74     };
 75 
 76     $.extend({
 77 
 78          /**
 79          * @param {object} options Configuration options
 80          * @return {Checkbox} Checkbox instance
 81          * @function
 82          * @name checkbox
 83          * @memberOf $.prototype 
 84          * @config {element} el Container element
 85          * @example #checkbox1
 86          *      %input{:type=>"checkbox"}
 87          *      %label Option 1
 88          *
 89          * :javascript
 90          *      $("#checkbox1").checkbox();
 91          */
 92         checkbox: function (options) {
 93             if (typeof options !== "object") {
 94                 options = {};
 95             }
 96             options.el = this.el;
 97             return new Checkbox(options);
 98         }
 99     });
