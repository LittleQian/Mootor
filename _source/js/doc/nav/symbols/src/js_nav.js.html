 10 (function ($) {
 11     "use strict";
 12 
 13     var fullWidth,
 14         Nav,
 15 
 16     fullWidth = function (el) {
 17         el.style.width = $.view.clientW + "px";
 18     };
 19      
 20      /**
 21      * Navigation object
 22      * @class
 23      * @return {Nav} Nav object
 24      * @param {object} options  Configuration options
 25      * @property {element} el Container element
 26      * @property {integer} height Container height
 27      * @property {integer} x Current position on X axis
 28      * @property {integer} y Current position on Y axis
 29      * @property {integer} current Index of active panel
 30      * @property {integer} back Index of previous active panel
 31      * @property {integer} width Nav container width
 32      * @property {integer} height Nav container height
 33      * @property {boolean} isMoving True if Nav container is moving
 34      * @property {integer} direction Direction for Nav movement
 35      * @property {array} [history] Navigation history
 36      * @property {integer} count Items count
 37      */
 38     Nav = function (options) {
 39 
 40         var i,
 41             /**
 42              * Navigation item
 43              * @ignore
 44              */
 45             item = {
 46                 anchors: [],
 47                 el: undefined,
 48                 height: 0,
 49                 hidden: []
 50             }
 51             Nav;
 52 
 53         this.el = options.el;
 54         this.navClass = options.nav_class !== undefined ? options.nav_class : "nav";
 55         this.itemClass = options.item_class !== undefined ? options.item_class : "panel";
 56         this.headerId = options.header_id !== undefined ? options.header_id : "header";
 57         this.hiddenClass = options.hidden_class !== undefined ? options.hidden_class : "hidden";
 58         this.margin = options.item_margin !== undefined ? options.item_margin : 5;
 59         this.width = options.width !== undefined ? options.width : $.view.clientW;
 60         this.height = options.height !== undefined ? options.height : $.view.clientH;
 61         this.x = 0;
 62         this.y = 0;
 63         this.current = 0;
 64         this.back = 0;
 65         this.items = [];
 66         this.header = this.header.init(this);
 67         this.isMoving = false;
 68         this.direction = 0;
 69         this.history = [];
 70 
 71         Nav = this.el.getElementsByClassName(this.itemClass);
 72 
 73         this.count = Nav.length;
 74 
 75         for (i = Nav.length; i--;) {
 76             this.items[i] = {el: Nav[i]};
 77             item = this.items[i];
 78             item.anchors = item.el.getElementsByClassName(this.navClass);
 79             item.hidden = item.el.getElementsByClassName(this.hiddenClass);
 80         }
 81 
 82         $(this.el).onDragMove(this);
 83         $(this.el).onDragEnd(this);
 84 
 85         if (document.body.style.overflow !== "hidden") {
 86             document.body.style.overflow = "hidden";
 87         }
 88 
 89         this.init();
 90 
 91         return this;
 92 
 93     };
 94 
 95     Nav.prototype = {
 96 
 97         /**
 98          * Gesture handler
 99          * @private
100          */
101         handleGesture: function (gesture) {
102             switch (gesture.type) {
103             case "dragMove":
104                 this.move(gesture);
105                 break;
106             case "dragEnd":
107                 this.check(gesture);
108                 break;
109             }
110         },
111 
112         /**
113          * Initialize header
114          * @private
115          */
116         header: {
117             init: function (panel) {
118                 var header = {};
119                 header.el = document.getElementById(panel.headerId);
120                 if (header.el) {
121                     panel.nav(header);
122                     fullWidth(header.el);
123                     panel.top = header.el.offsetHeight;
124                     panel.height = $.view.clientH - panel.top;
125                     panel.el.style.marginTop = panel.top + "px";
126                     return header;
127                 }
128             }
129         },
130 
131         /**
132          * Initialize navigation area
133          * @private
134          */
135         nav: function (obj) {
136             obj.anchors = obj.el.getElementsByClassName(this.navClass);
137         },
138 
139         /**
140          * Initialize Nav
141          * @private
142          */
143         init: function () {
144 
145             var anchorCallback,
146                 j,
147                 i,
148                 Nav = this,
149                 panel,
150                 setActive,
151                 headerAnchor,
152                 goBack,
153                 clickcb;
154 
155             anchorCallback = function (gesture) {
156                 Nav.direction = 0;
157                 $(gesture.el).removeClass("active");
158                 if (Nav.isMoving === false) {
159                     Nav.set(gesture.el.rel);
160                 }
161                 return false;
162             };
163 
164             clickcb = function () {
165                 return false;
166             };
167 
168             setActive = function (gesture) {
169                 $(gesture.el).setClass("active");
170             };
171 
172             // Reset styles and set anchor links
173             for (i = this.count; i--;) {
174 
175                 panel = this.items[i];
176 
177                 if (this.width === undefined) {
178                     fullWidth(panel.el);
179                 } else {
180                     panel.el.style.width = this.width + "px";
181                 }
182 
183                 panel.el.style.overflow = 'hidden';
184 
185                 if (i > 0) {
186                     panel.el.style.left = -((this.width + this.margin) * 4) + "px";
187                     panel.el.style.top = "0px";
188                 } else {
189                     panel.el.style.left = "0px";
190                 }
191 
192                 panel.height = panel.el.offsetHeight;
193                 if (this.height > panel.height) {
194                     panel.el.style.height = this.height + "px";
195                     panel.height = this.height;
196                 }
197 
198                 for (j = panel.anchors.length; j--;) {
199                     if (panel.anchors[j].rel !== "") {
200                         $(panel.anchors[j]).onTapStart(setActive);
201                         panel.anchors[j].onclick = clickcb;
202                         $(panel.anchors[j]).onTapEnd(anchorCallback);
203                     }
204                 }
205             }
206 
207             if (this.header) {
208                 goBack = function () {
209                     Nav.goBack();
210                 };
211                 for (i = this.header.anchors.length; i--;) {
212                     headerAnchor =  this.header.anchors[i];
213                     if (headerAnchor.rel === "back") {
214                         $(headerAnchor).onTapEnd(goBack);
215                         headerAnchor.onclick = clickcb;
216                     }
217                 }
218             }
219 
220             return this;
221 
222         },
223 
224         /**
225          * Move panel
226          * @private
227          */
228         move: function (gesture) {
229             var panel =  this.items[this.current];
230             if (gesture.isDraggingY !== 0 && panel.movable !== false) {
231                 this.isMoving = true;
232                 this.y = this.y + (gesture.y - gesture.lastY);
233                 this.translate({
234                     el: panel.el,
235                     y: this.y
236                 });
237             }
238 
239         },
240 
241         /**
242          * Check movement
243          * @private
244          */
245         check: function (gesture) {
246             var panel = this.items[this.current],
247                 maxdist = panel.height - this.height,
248                 cb,
249                 i;
250 
251             if (gesture.isDraggingY !== 0) {
252 
253                 this.isMoving = false;
254                 // Bounce back
255                 if (this.y >= 0 || maxdist < -this.y) {
256                     if (this.y > 0) {
257                         this.y = 0;
258                     } else {
259                         this.y = -(panel.height - this.height);
260                     }
261                     for (i = panel.anchors.length; i--;) {
262                         $(panel.anchors[i]).removeClass("active");
263                     }
264                     this.translate({
265                         y: this.y,
266                         el: panel.el,
267                         duration: 0.5
268                     });
269                 }
270 
271             }
272 
273         },
274 
275         /**
276          * Load current panel
277          * @private
278          */
279         load: function () {
280 
281             var panel,
282                 cb,
283                 back,
284                 show = this.show,
285                 translate = this.translate,
286                 positionX,
287                 container = this.el;
288 
289             panel = this.items[this.current];
290             back = this.items[this.back];
291 
292             this.hide(panel);
293             this.hide(back);
294             $(panel.el).show();
295 
296             cb = function () {
297                 show(panel);
298                 $(back.el).hide();
299             };
300 
301             positionX = this.width + this.margin;
302 
303             if (this.current !== 0) {
304                 if (this.back === 0) {
305                     panel.el.style.left = positionX + "px";
306                 } else {
307                     translate({el: container});
308                     back.el.style.left = "0px";
309                     if (this.direction !== 0) {
310                         positionX = -positionX;
311                     }
312                     panel.el.style.left = positionX + "px";
313                 }
314             } else if (this.back !== 0) {
315                 translate({el: container, x: -positionX});
316                 back.el.style.left = positionX + "px";
317                 panel.el.style.left = "0px";
318                 positionX = 0;
319             }
320 
321             if (this.side === 1) {
322                 this.side = 0;
323             } else {
324                 this.side = 1;
325             }
326 
327             window.setTimeout(function () {
328                 translate({
329                     el: container,
330                     duration: 0.5,
331                     x: -positionX,
332                     callback: cb
333                 });
334             }, 10);
335         },
336 
337         /**
338          * Set current panel
339          * @param {string} id Nav id
340          * @example nav.set("home");
341          */
342         set: function (id) {
343 
344             var i;
345 
346             // Get panel by id and load it
347             for (i = this.count; i--;) {
348                 if (this.items[i].el.id === id) {
349                     this.back = this.current;
350                     if (this.direction === 0) {
351                         this.get(id);
352                         this.history.push(this.current);
353                     }
354                     this.current = i;
355                     this.load();
356                 }
357             }
358 
359         },
360 
361         /**
362          * Get panel by id
363          * @param {string} id Nav id
364          * @private
365          * @example nav.get("home").height;
366          * @return {item} Navigation item
367          */
368         get: function (id) {
369             var i;
370             // Get panel by id and load it
371             for (i = this.count; i--;) {
372                 if (this.items[i].el.id === id) {
373                     return this.items[i];
374                 }
375             }
376         },
377 
378         /**
379          * Translate Nav
380          * @param {object} options
381          * @private
382          * @config {integer} duration Translate duration, in seconds
383          * @config {function} callback Function to call on move end
384          * @config {integer} x Active panel position on X axis
385          * @config {integer} y Active panel position on Y axis
386          * @config {integer} y Active panel position on Y axis
387          * @example item.translate({x: 100});
388          */
389         translate: function (options) {
390             if (options.duration === undefined) {
391                 options.duration = 0;
392             }
393             if (options.callback === undefined) {
394                 options.callback = function () {};
395             }
396             if (options.y === undefined) {
397                 options.y = 0;
398             }
399             if (options.x === undefined) {
400                 options.x = 0;
401             }
402             $(options.el).translate(
403                 {y: options.y, x: options.x},
404                 {transitionDuration: options.duration, callback: options.callback}
405             );
406         },
407 
408         /**
409          * Hide hidden content
410          * @param {Nav} panel
411          * @private
412          * @example nav.hide(nav.items[nav.current])
413          */
414         hide: function (panel) {
415             var i;
416             for (i = panel.hidden.length; i--;) {
417                 $(panel.hidden[i]).hide();
418             }
419         },
420 
421         /**
422          * Show hidden content
423          * @private
424          * @example nav.show();
425          */
426         show: function (panel) {
427             var i;
428             for (i = panel.hidden.length; i--;) {
429                 $(panel.hidden[i]).show();
430             }
431         },
432 
433         /**
434          * Go back on navigation history
435          * @example nav.goBack();
436          */
437         goBack: function () {
438             if (this.history.length > 0) {
439                 this.direction = -1;
440                 this.back = this.history.pop();
441                 this.set(this.items[this.back].el.id);
442             }
443         },
444 
445         /**
446          * Nav configuration
447          * @param {object} options
448          * @config {string} panel Nav id
449          * @config {boolean} movable Set to false to block panel
450          * @example nav.config({
451          *      panel: "geo",
452          *       movable: false
453          * });
454          */
455         config: function (options) {
456             var panel;
457             if (options.panel !== undefined) {
458                 panel = this.get(options.panel);
459                 if (options.movable !== undefined) {
460                     panel.movable = options.movable;
461                 }
462             }
463         }
464 
465     };

477          * @return {Nav} Nav object
478          * @config {string} navClass Navigation class name
479          * @config {string} itemClass Nav class name
480          * @config {string} hiddenClass Hidden content class name
481          * @config {string} headerId Header element id
482          * @example var nav = $("#myPanels").nav();
483          */
484         nav: function (options) {
485             if (typeof options !== "object") {
486                 options = {};
487             }
488             options.el = this.el;
489             return new Nav(options);
490         }
491     });
