  5  /** 
  6  * @class
  7  * @name $ 
  8  */
  9 
 10 var $ = window.$ || $;
 11 
 12 (function ($) {
 13 
 14     "use strict";
 15 
 16     var Map;
 17 
 18     /**
 19      * Map
 20      * @class
 21      * @param {element} element HTML container element
 22      * @property {element} element HTML container element
 23      */
 24     Map = function (element) {
 25 
 26         var success,
 27             error,
 28             marker,
 29             latlng;
 30 
 31         this.el = element;
 32 
 33         success = function (position) {
 34             var options,
 35                 map,
 36                 google = window.google || undefined,
 37                 stop;
 38 
 39             if (google !== undefined) {
 40 
 41                 latlng = new google.maps.LatLng(position.coords.latitude, position.coords.longitude);
 42 
 43                 options = {
 44                     zoom: 15,
 45                     center: latlng,
 46                     mapTypeControl: false,
 47                     navigationControlOptions: {style: google.maps.NavigationControlStyle.SMALL},
 48                     mapTypeId: google.maps.MapTypeId.ROADMAP
 49                 };
 50 
 51                 map = new google.maps.Map(document.getElementById(element.id), options);
 52 
 53                 marker = new google.maps.Marker({
 54                     position: latlng,
 55                     map: map,
 56                     title: "You are here! (at least within a " + position.coords.accuracy + " meter radius)"
 57                 });
 58 
 59             }
 60 
 61             return this;
 62 
 63         };
 64 
 65         error = function () {
 66             element.innerHTML = "error";
 67         };
 68 
 69         if (window.navigator.geolocation) {
 70             window.navigator.geolocation.getCurrentPosition(success, error);
 71         } else {
 72             this.el.innerHTML =  'not supported';
 73         }
 74 
 75     };
 76 
 77     $.extend({
 78         /**          
 79          * Map
 80          * @name map
 81          * @function
 82          * @memberOf $.prototype
 83          * @example $("#map").map();
 84          * @return {Map} Geolocated map
 85          */
 86         map: function () {
 87             return new Map(this.el);
 88         }
 89 
 90     });
 91 
 92 }($));
