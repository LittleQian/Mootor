392     addGesture = function (options) {
393         var gestureList = $.gestures.list,
394             type = options.type,
395             fn = options.fn,
396             callback = options.callback,
397             key = createKey(fn.el);

399         if (gestureList[key] === undefined) {
400             gestureList[key] = {
401                 event: []
402             };
403             // Bind listeners only once
404             fn.bind("mousedown", fn);
405             fn.bind("mouseup", fn);
406             fn.bind("touchstart", fn);
407             fn.bind("touchend", fn);
408         }
409 
410         if (gestureList[key].event[type] === undefined) {
411             gestureList[key].event[type] = [];
412         }
413 
414         gestureList[key].event[type].push(callback);

425                     callbacks[i].handleGesture(info);
426                 } else {
427                     callbacks[i](info);

429             }

558         // Handler to detect gestures and fire callbacks        
559         handleEvent: function (e) {
560             var key = createKey(this.el),
561                 info = {
562                     el: this.el
563                 },
564                 gesture = $.gestures.list[key],
565                 date = new Date(),
566                 clientX,
567                 clientY;
568 
569             e.preventDefault();
570 
571             if (e.clientX || e.clientY) {
572                 // Mouse
573                 clientX = e.clientX;
574                 clientY = e.clientY;
575             } else {
576                 // Touch
577                 // FIXME CHECK
578                 try {
579                     clientX = e.touches[0].clientX;
580                     clientY = e.touches[0].clientY;
581                 } catch (error) {}
582             }
583 
584             // TapStart
585             if (e.type === "mousedown" || e.type === "touchstart") {
586 
587                 this.bind("mousemove", this);
588                 this.bind("touchmove", this);
589 
590                 gesture.event.time = date.getTime();
591                 gesture.event.isDraggingY = 0;
592                 gesture.event.mousedown = true;
593                 gesture.event.tapped = false;
594                 gesture.event.startX = clientX;
595                 gesture.event.startY = clientY;
596 
597                 window.setTimeout(function () {
598                     // TapHold
599                     if (gesture.event.mousedown === true) {
600                         info.type = "tapHold";
601                         fire(info, gesture.event.onTapHold);
602                     }
603                 }, 500);
604 
605                 if (gesture.event.onTapStart !== undefined) {
606                     // TapStart
607                     info.type = "tapStart";
608                     fire(info, gesture.event.onTapStart);
609                 }
610             }
611 
612             if (e.type === "mousemove" || e.type === "touchmove") {
613 
614                 info.lastY = gesture.event.y;
615                 info.lastX = gesture.event.x;
616                 gesture.event.y = info.y = clientY;
617                 gesture.event.x = info.x = clientX;
618                 info.distanceFromOriginY = clientY - gesture.event.startY;
619                 info.distanceFromOriginX = clientX - gesture.event.startX;
620 
621                 if (gesture.event.isDraggingY === 0) {
622                     // DragStart
623                     if ((clientY - gesture.event.startY) > 10) {
624                         gesture.event.isDraggingY = 1;
625                         info.type = "dragStart";
626                         fire(info, gesture.event.onDragStart);
627                     } else if ((clientY - gesture.event.startY) < -10) {
628                         gesture.event.isDraggingY = -1;
629                         info.type = "dragStart";
630                         fire(info, gesture.event.onDragStart);
631                     }
632                 } else {
633                     // DragMove
634                     info.type = "dragMove";
635                     fire(info, gesture.event.onDragMove);
636                 }
637             }
638 
639             if (e.type === "mouseup" || e.type === "touchend") {
640 
641                 if (gesture.event.tapped === false) {
642                     this.unbind("mousemove", this);
643                     this.unbind("touchmove", this);
644                     gesture.event.tapped = true;
645                     info.time = date.getTime() - gesture.event.time;
646                     gesture.event.mousedown = false;
647                 }
648 
649                 if (gesture.event.isDraggingY !== 0) {
650                     // DragEnd
651                     info.type = "dragEnd";
652                     gesture.event.isDraggingY = 0;
653                     fire(info, gesture.event.onDragEnd);
654 
655                 } else {
656                     // TapEnd
657                     info.type = "tapEnd";
658                     fire(info, gesture.event.onTapEnd);
659                 }
660             }
661 
662         }
