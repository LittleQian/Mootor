 36 	var Moo = function (query) {
 37 		var qtype = typeof query,
 38 			el;
 39 
 40         // Get element from query
 41         if (qtype === "string") {
 42 
 43             if (query.indexOf("#") > -1) {
 44                 query = query.replace("#", "");
 45                 el = document.getElementById(query);
 46 
 47             } else if (query.indexOf(".") > -1) {
 48                 query = query.replace(".", "");
 49                 el = document.getElementsByClassName(query);
 50 
 51             }
 52         } else if (qtype === "object") {
 53         el = query;
 54         }
 55 
 56         // Instance properties
 57         this.el = (function () {
 58             return el;
 59         }());
 60         this.query = (function () {
 61             return query;
 62         }());
 63 
 64 		return this;
 65 	};

306     var createKey,
307         addGesture,
308         fire;

310      $.extend({
311         /**
312          * Gestures
313          *
314          * @class
315          * @name gestures
316          * @memberOf $
317          * @property {integer} x Position on X axis
318          * @property {integer} y Position on Y axis
319          * @property {integer} startX Position on X axis at the start of the gesture
320          * @property {integer} endX Position on X axis at the end of the gesture
321          * @property {integer} startY Position on Y axis at the start of the gesture
322          * @property {integer} endY Position on Y axis at the end of the gesture
323          * @property {boolean} isDraggingY Return true when is dragging on Y axis
324          * @property {boolean} mousedown Return true when mouse or touch is down
325          * @property {boolean} tapped Return true when a onTap was fired
326          * @property {integer} time Time between last 2 touchs
327          * @property {element} el Element binded to gesture
328          */
329         gestures: {
330             list: []
331         }
332     }, $);
333 
334     // Create key for element
335     createKey = function (el) {
336 
337         if (el.id !== "") {
338             return el.id;
339         } else if (el.rel !== undefined) {
340             return el.rel;
341         } else if (typeof el === "object") {
342             return el;
343         }
344     };
345 
346     addGesture = function (options) {
347         var gestureList = $.gestures.list,
348             type = options.type,
349             fn = options.fn,
350             callback = options.callback,
351             key = createKey(fn.el);
352 
353         if (gestureList[key] === undefined) {
354             gestureList[key] = {
355                 event: []
356             };
357             // Bind listeners only once
358             fn.bind("mousedown", fn);
359             fn.bind("mouseup", fn);
360             fn.bind("touchstart", fn);
361             fn.bind("touchend", fn);
362         }
363 
364         if (gestureList[key].event[type] === undefined) {
365             gestureList[key].event[type] = [];
366         }
367 
368         gestureList[key].event[type].push(callback);
369 
370     };
371 
372     // Fire callbacks
373     fire = function (info, callbacks) {
374         var i;
375 
376         if (callbacks !== undefined) {
377             for (i = 0; i < callbacks.length; i++) {
378                 if (callbacks[i].handleGesture !== undefined) {
379                     callbacks[i].handleGesture(info);
380                 } else {
381                     callbacks[i](info);
382                 }
383             }
384         }
385     };

494         // Handler to detect gestures and fire callbacks        
495         handleEvent: function (e) {
496             var key = createKey(this.el),
497                 info = {
498                     el: this.el
499                 },
500                 gesture = $.gestures.list[key],
501                 date = new Date(),
502                 clientX,
503                 clientY;
504 
505             e.preventDefault();
506 
507             if (e.clientX || e.clientY) {
508                 // Mouse
509                 clientX = e.clientX;
510                 clientY = e.clientY;
511             } else {
512                 // Touch
513                 // FIXME CHECK
514                 try {
515                     clientX = e.touches[0].clientX;
516                     clientY = e.touches[0].clientY;
517                 } catch (error) {}
518             }
519 
520             // TapStart
521             if (e.type === "mousedown" || e.type === "touchstart") {
522 
523                 this.bind("mousemove", this);
524                 this.bind("touchmove", this);
525 
526                 gesture.event.time = date.getTime();
527                 gesture.event.isDraggingY = 0;
528                 gesture.event.mousedown = true;
529                 gesture.event.tapped = false;
530                 gesture.event.startX = clientX;
531                 gesture.event.startY = clientY;
532 
533                 window.setTimeout(function () {
534                     // TapHold
535                     if (gesture.event.mousedown === true) {
536                         info.type = "tapHold";
537                         fire(info, gesture.event.onTapHold);
538                     }
539                 }, 500);
540 
541                 if (gesture.event.onTapStart !== undefined) {
542                     // TapStart
543                     info.type = "tapStart";
544                     fire(info, gesture.event.onTapStart);
545                 }
546             }
547 
548             if (e.type === "mousemove" || e.type === "touchmove") {
549 
550                 info.lastY = gesture.event.y;
551                 info.lastX = gesture.event.x;
552                 gesture.event.y = info.y = clientY;
553                 gesture.event.x = info.x = clientX;
554                 info.distanceFromOriginY = clientY - gesture.event.startY;
555                 info.distanceFromOriginX = clientX - gesture.event.startX;
556 
557                 if (gesture.event.isDraggingY === 0) {
558                     // DragStart
559                     if ((clientY - gesture.event.startY) > 10) {
560                         gesture.event.isDraggingY = 1;
561                         info.type = "dragStart";
562                         fire(info, gesture.event.onDragStart);
563                     } else if ((clientY - gesture.event.startY) < -10) {
564                         gesture.event.isDraggingY = -1;
565                         info.type = "dragStart";
566                         fire(info, gesture.event.onDragStart);
567                     }
568                 } else {
569                     // DragMove
570                     info.type = "dragMove";
571                     fire(info, gesture.event.onDragMove);
572                 }
573             }
574 
575             if (e.type === "mouseup" || e.type === "touchend") {
576 
577                 if (gesture.event.tapped === false) {
578                     this.unbind("mousemove", this);
579                     this.unbind("touchmove", this);
580                     gesture.event.tapped = true;
581                     info.time = date.getTime() - gesture.event.time;
582                     gesture.event.mousedown = false;
583                 }
584 
585                 if (gesture.event.isDraggingY !== 0) {
586                     // DragEnd
587                     info.type = "dragEnd";
588                     gesture.event.isDraggingY = 0;
589                     fire(info, gesture.event.onDragEnd);
590 
591                 } else {
592                     // TapEnd
593                     info.type = "tapEnd";
594                     fire(info, gesture.event.onTapEnd);
595                 }
596             }
597 
598         }
599     });

610     $.extend({

614         /**
615          * Translate element, using GPU acceleration when available
616          * @param {object} positions Axis positions
617          * @param {object} options Options
618          * @config {integer} transitionDuration Duration of transition (in seconds)
619          * @example $.Fx.translate($("#myDiv", {10,20}, {tansitionDuration: .5}));
620          */
621         translate: function (positions, options) {
622 
623             var x_pos = positions.x,
624                 y_pos = positions.y,
625                 tduration;
626                
627             tduration = options.transitionDuration;
628             this.el.style.transitionProperty = "webkit-transform";
629 
630             if (tduration !== undefined && tduration > 0) {
631                 this.el.style.webkitTransitionDuration = tduration + "s";
632                 this.el.style.webkitTransitionTimingFunction = "ease-out";
633             } else {
634                 if (this.el.style.webkitTransitionDuration !== "") {
635                     this.cleanFx();
636                 }
637             }
638 
639             this.el.style.webkitTransform = "translate3d(" + x_pos + "px," + y_pos + "px, 0)";
640 
641             if (options.callback) {
642                 window.setTimeout(options.callback, tduration * 1000);
643             }
644 
645         },
646 
647         /**
648          * Clean element transform styles
649          * @param {element} el Element
650          * @example $.Fx.clean($("#myDiv");
651          */
652         cleanFx: function () {
653             this.el.style.webkitTransitionDuration = "";
654             this.el.style.webkitTransitionTimingFunction = "";
655         }
